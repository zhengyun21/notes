2. 实现一个多线程应用：我来实现这个多线程应用。根据你的描述，需求如下：
- 数组长度为 n，元素为 1 到 n
- 线程A：如果元素能被3整除（但不能被5整除？按描述理解），输出"A"
- 线程B：如果元素能被5整除，输出"B"
- 线程C：如果能同时被5和3整除（即被15整除），输出"C"
- 线程D（主线程？）：如果元素既不能被3也不能被5整除，输出元素本身
这里我假设优先级是：先判断能否被15整除（C），再判断单独整除3或5，最后判断都不能整除。
```java
import java.util.concurrent.*;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

public class OrderedMultiThreadChecker {
    private final int[] array;
    private final ExecutorService executor;
    private final Map<Integer, String> results = new ConcurrentHashMap<>();
    private final CountDownLatch latch;
    
    public OrderedMultiThreadChecker(int n) {
        this.array = new int[n];
        for (int i = 0; i < n; i++) {
            array[i] = i + 1;
        }
        this.executor = Executors.newFixedThreadPool(4);
        this.latch = new CountDownLatch(n);
    }
    
    public void checkNumber(int num) {
        String result;
        if (num % 15 == 0) {
            result = "C";
        } else if (num % 3 == 0) {
            result = "A";
        } else if (num % 5 == 0) {
            result = "B";
        } else {
            result = String.valueOf(num);
        }
        results.put(num, result);
        latch.countDown();
    }
    
    public void startProcessing() {
        for (int num : array) {
            final int n = num;
            executor.submit(() -> checkNumber(n));
        }
        executor.shutdown();
        try {
            latch.await(); // 等待所有任务完成
            // 按顺序输出
            for (int i = 1; i <= array.length; i++) {
                System.out.print(results.get(i) + " ");
                if (i % 20 == 0) System.out.println(); //每20个换行
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
    
    public static void main(String[] args) {
        int n = 30;
        System.out.println("数组长度: " + n);
        System.out.print("结果: ");
        OrderedMultiThreadChecker checker = new OrderedMultiThreadChecker(n);
        checker.startProcessing();
    }
}

1 - 输入一个数组，元素是**字符串**（包括 `001` 这种会被解析为数字的情况）
- 判断每个元素的“有效长度”：
    - 如果是**纯数字**（无论是 `1`、`001`、`123`），输出**数字本身的数值大小**（比如 `001` 输出 1，`123` 输出 123）
    - 如果是**字符串**或**数字与字符混合**，输出**字符串的长度**
- 最后找出**最长**的那个元素输出
示例：
- 数组：`["1", "abc", "a1b2c3", 001, "hello", "1234"]`
- 输出：6
- 数组：`[1, 001, 01, 001]`
- 输出：1
- ```java
import java.util.*;

public class StringLengthEvaluator {
    
    /**
     * 判断字符串是否为纯数字（包括 "001" 这种情况）
     */
    private static boolean isPureNumeric(String str) {
        if (str == null || str.isEmpty()) {
            return false;
        }
        // 检查每个字符是否都是数字
        for (char c : str.toCharArray()) {
            if (!Character.isDigit(c)) {
                return false;
            }
        }
        return true;
    }
    
    /**
     * 获取元素的"有效长度"
     * @param element 可以是String、Integer或其他类型
     * @return 有效长度值
     */
    private static int getEffectiveLength(Object element) {
        String str = element.toString();
        
        // 判断是否为纯数字
        if (isPureNumeric(str)) {
            // 去除前导零并转换为数字
            try {
                // 使用Integer.parseInt会自动处理前导零
                int numericValue = Integer.parseInt(str);
                return numericValue;
            } catch (NumberFormatException e) {
                // 如果数字太大，返回字符串长度
                return str.length();
            }
        } else {
            // 非纯数字，返回字符串长度
            return str.length();
        }
    }
    
    /**
     * 找出最长元素并返回其有效长度
     */
    public static int findMaxEffectiveLength(Object[] array) {
        if (array == null || array.length == 0) {
            return 0;
        }
        
        int maxLength = 0;
        
        for (Object element : array) {
            int effectiveLength = getEffectiveLength(element);
            System.out.println("元素: " + element + " (类型: " + 
                             element.getClass().getSimpleName() + 
                             ") → 有效长度: " + effectiveLength);
            
            if (effectiveLength > maxLength) {
                maxLength = effectiveLength;
            }
        }
        
        return maxLength;
    }
    
    public static void main(String[] args) {
        // 示例1：混合类型数组
        System.out.println("=== 示例1 ===");
        Object[] array1 = {"1", "abc", "a1b2c3", 001, "hello", "1234"};
        int result1 = findMaxEffectiveLength(array1);
        System.out.println("最长有效长度: " + result1);
        System.out.println();
        
        // 示例2：全是数字数组
        System.out.println("=== 示例2 ===");
        Object[] array2 = {1, 001, 01, 001};
        int result2 = findMaxEffectiveLength(array2);
        System.out.println("最长有效长度: " + result2);
        System.out.println();
        
        // 额外测试示例
        System.out.println("=== 额外测试 ===");
        Object[] array3 = {"999", "1000", "abc123", "000", "hello world"};
        int result3 = findMaxEffectiveLength(array3);
        System.out.println("最长有效长度: " + result3);
    }
}
